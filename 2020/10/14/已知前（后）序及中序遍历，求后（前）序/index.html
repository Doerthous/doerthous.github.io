已知前（后）序及中序遍历，求后（前）序

算法题

题目

已知二叉树的前/后序及中序遍历，求后/前序遍历。

典型题目：已知二叉树中序遍历为ADEFGHMZ，后序遍历为AEFDHZMG，求前序遍历。

解题思路

前中后序列遍历的基本概念 => 前中后序遍历的规律 => 算法思路

前中后序遍历的基本概念

  P
 / \
L   R

前序（pre-order）: PLR
中序（in-order）: LPR
后序（post-order）: LRP

序指的是根节点在左右子树中的位置：根左右、左根右、左右根。

前中后序列遍历的规律

  • 前序遍历：如果在同一颗树内，最前边的节点是根节点。
  • 中序遍历：如果在同一颗树内，根左边的是左子树，根右边的是右子树。
  • 后序遍历：如果在同一颗树内，最后边的节点是根节点。

算法思路

  1. 通过前序（后序）遍历，找到树的根节点。
  2. 通过根节点及中序遍历，划分左右子树。
  3. 对左右子树递归进行步骤1、2，直到子树划分完毕。

总结：找根节点、划分左右子树。

人型电脑实验

已知二叉树中序遍历为ADEFGHMZ，后序遍历为AEFDHZMG，求前序遍历。

    r = pre[0]
    find in[i] == r
    r->left = bt_solve_from_inpreorder(in, i, pre+1, i)
    r->right = bt_solve_from_inpreorder(in+i+1, in_len-i-1, 
        pre+1+i, pre_len-i-1)
    return r;

C

#include <stdio.h>
#include <stdlib.h>
#include <assert.h>

typedef struct bt
{
    char key;
    struct bt *left;
    struct bt *right;
} bt_t;

void bt_preorder(bt_t *bt, void (*op)(bt_t *))
{
    if (bt) {
        op(bt);
        bt_preorder(bt->left, op);
        bt_preorder(bt->right, op);
    }
}

void bt_inorder(bt_t *bt, void (*op)(bt_t *))
{
    if (bt) {
        bt_inorder(bt->left, op);
        op(bt);
        bt_inorder(bt->right, op);
    }
}

void bt_postorder(bt_t *bt, void (*op)(bt_t *))
{
    if (bt) {
        bt_postorder(bt->left, op);
        bt_postorder(bt->right, op);
        op(bt);
    }
}

bt_t *
bt_solve_from_inpostorder(
    const char *in, int in_len, const char *post, int post_len
)
{
    if (post_len > 0) {
        int i;
        bt_t *r = (bt_t *)malloc(sizeof(bt));
        assert(r);

        r->key = post[post_len-1];
        for (i = 0; i < in_len && in[i] != r->key; ++i);
        r->left = bt_solve_from_inpostorder(in, i, post, i);
        r->right = bt_solve_from_inpostorder(in+i+1, in_len-i-1, 
            post+i, post_len-i-1);
        return r;
    }

    return NULL;
}

bt_t *
bt_solve_from_inpreorder(
    const char *in, int in_len, const char *pre, int pre_len
)
{
    if (pre_len > 0) {
        int i;
        bt_t *r = (bt_t *)malloc(sizeof(bt));
        assert(r);

        r->key = pre[0];
        for (i = 0; i < in_len && in[i] != r->key; ++i);
        r->left = bt_solve_from_inpreorder(in, i, pre+1, i);
        r->right = bt_solve_from_inpreorder(in+i+1, in_len-i-1, 
            pre+1+i, pre_len-i-1);
        return r;
    }

    return NULL;
}

void bt_node_print(bt_t *bt)
{
    printf("%c", bt->key);
}

int main()
{
    const char *in = "ADEFGHMZ";
    const char *post = "AEFDHZMG";
    const char *pre = "GDAFEMHZ";

    bt_t *bt = bt_solve_from_inpostorder(in, 8, post, 8);
    bt_preorder(bt, bt_node_print);
    bt_postorder(bt, (void (*)(bt_t *))free);

    printf("\n");
    
    bt = bt_solve_from_inpreorder(in, 8, pre, 8);
    bt_postorder(bt, bt_node_print);
    bt_postorder(bt, (void (*)(bt_t *))free);

    return 0;
}
