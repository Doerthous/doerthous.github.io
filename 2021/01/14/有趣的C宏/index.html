1 变量后添加行号

想法来源：在一些自动生成变量而又不会直接使用到该变量的情况。

#define CONCAT_(x,y) x##y
#define CONCAT(x,y) CONCAT_(x,y)
#define VAR_LINE(v) CONCAT(v,__LINE__)

int main()
{
    int VAR_LINE(a);
    return 0;
}

预编译后：

int main()
{
    int a7;
    return 0;
}

用法举例：

#include <stdio.h>
#include <stdint.h>
#include <string.h>

#define CONCAT_(x,y) x##y
#define CONCAT(x,y) CONCAT_(x,y)
#define VAR_LINE(v) CONCAT(v,__LINE__)

#define alloc_from_stack(pp, size) \
    uint8_t VAR_LINE(stack_memory)[size]; \
    *(pp) = VAR_LINE(stack_memory);

int main()
{
    uint8_t *p;

    alloc_from_stack(&p, 12);
    memcpy(p, "Hello world\0", 12);
    printf("%s\n", p);

    return 0;
}

2 List foreach

来源：如何在遍历list的同时删除其中的节点，又不暴露list内部的next指针。

#include <stdio.h>

// list.h
typedef struct list
{
    struct list *next;
} list_t;

#define list_append(list, node) do \
{ \
    list_t **it = &(((list_t *)(list))->next); \
    ((list_t *)node)->next = NULL; \
    while (*it) { \
        it = &((*it)->next); \
    } \
    (*it) = (list_t *)node; \
} while (0)

#define list_foreach_remove(node) list_foreach_rm = 1; *node = (*(node))->next;
#define list_foreach(type, name, list, do_something) \
    for (type **name = (type **)&((list)->next); *name; ) { \
        int list_foreach_rm = 0; \
        do_something \
        if (*name && !list_foreach_rm) { \
            name = &((*name)->next); \
        } \
    }

#define LIST_INIT(list) (list)->next = NULL;

// something.h
typedef struct int_node
{
    struct int_node *next;
    int val;
} int_node_t;

// main.c
int main(int argc, char const *argv[])
{
    list_t list;
    int_node_t n[5];

    LIST_INIT(&list);

    for (int i = 0; i < 5; ++i) {
        n[i].val = i;
        list_append(&list, &n[i]);
    }

    // 输出：0 1 2 3 4
    list_foreach(int_node_t, item, &list, {
        printf("%d ", (*item)->val);
    });

    // 删除0，1，4节点
    list_foreach(int_node_t, item, &list, {
        if ((*item)->val == 0 || (*item)->val == 1 || (*item)->val == 4) {
            list_foreach_remove(item);
        }
    });

    printf("\n");

    // 输出：2 3
    list_foreach(int_node_t, item, &list, {
        printf("%d ", (*item)->val);
    });

    return 0;
}

/* 完整输出：
0 1 2 3 4
2 3
*/

3 协程

#define c_start() static int __c_break_point;switch(__c_break_point)default:{
#define c_end() } return 0;
#define c_yield() do{__c_break_point=__LINE__;return 1;case __LINE__:;}while(0)
#define c_yield_val(val) do { *c_yval = (val); c_yield(); } while (0)
#define c_call(c_func, ...) do \
{ \
    case __LINE__: if (c_func(__VA_ARGS__)) { \
        __c_break_point = __LINE__; return 1; \
    } \
} while (0)

#include <stdio.h>
int c_level_3() {
    c_start();
    printf("l3: c_yield()\n"); c_yield();
    c_end();
}
int c_level_2(int *c_yval) {
    c_start();
    printf("l2: c_yield_val(201)\n"); c_yield_val(201);
    c_call(c_level_3);
    printf("l2: c_yield_val(202)\n"); c_yield_val(202);
    c_end();
}
int c_level_1(int *c_yval) {
    c_start();
    printf("l1: c_yield_val(101)\n"); c_yield_val(101);
    printf("l1: c_yield()\n"); c_yield();
    static int i = 3;
    while(--i) {
        printf("l1: c_yield_val(i*4+100), i=%d\n", i); c_yield_val(i*4+100);
    }
    c_call(c_level_2, c_yval);
    c_end();
}
int main(int argc, char const *argv[]) {
    while (c_level_1(&argc))
        printf("yield: %d\n", argc);
    return 0;
}
